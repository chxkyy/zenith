CREATE TABLE public.t_sys_org (
	id bigserial NOT NULL, -- 主键ID，自增序列
	parent_id int8 DEFAULT 0 NULL, -- 父级组织ID，0表示顶级组织
	"name" varchar(100) NOT NULL, -- 组织/部门名称
	sort int4 DEFAULT 0 NULL, -- 排序号，数值越小越靠前
	status int4 DEFAULT 1 NULL, -- 组织状态：1-启用，0-禁用
	created_time timestamp DEFAULT CURRENT_TIMESTAMP NULL, -- 创建时间
	create_user_id int8 NULL, -- 创建人ID，关联 t_sys_user.id
	update_user_id int8 NULL, -- 最后更新人ID，关联 t_sys_user.id
	update_time timestamp DEFAULT CURRENT_TIMESTAMP NULL, -- 最后更新时间
	CONSTRAINT t_sys_org_pkey PRIMARY KEY (id)
);
COMMENT ON TABLE public.t_sys_org IS '系统组织架构表，定义组织/部门的树形结构';

-- Column comments

COMMENT ON COLUMN public.t_sys_org.id IS '主键ID，自增序列';
COMMENT ON COLUMN public.t_sys_org.parent_id IS '父级组织ID，0表示顶级组织';
COMMENT ON COLUMN public.t_sys_org."name" IS '组织/部门名称';
COMMENT ON COLUMN public.t_sys_org.sort IS '排序号，数值越小越靠前';
COMMENT ON COLUMN public.t_sys_org.status IS '组织状态：1-启用，0-禁用';
COMMENT ON COLUMN public.t_sys_org.created_time IS '创建时间';
COMMENT ON COLUMN public.t_sys_org.create_user_id IS '创建人ID，关联 t_sys_user.id';
COMMENT ON COLUMN public.t_sys_org.update_user_id IS '最后更新人ID，关联 t_sys_user.id';
COMMENT ON COLUMN public.t_sys_org.update_time IS '最后更新时间';

CREATE TABLE public.t_sys_user (
	id bigserial NOT NULL, -- 主键ID，自增序列
	username varchar(50) NOT NULL, -- 用户名，唯一标识，用于登录
	email varchar(100) NOT NULL, -- 电子邮箱
	status int4 DEFAULT 1 NOT NULL, -- 用户状态：1-正常，0-禁用
	"role" varchar(50) NULL, -- 角色编码，关联角色表(如 ROLE_ADMIN / ROLE_USER)
	org_name varchar(100) NULL, -- 所属组织名称，关联组织表
	created_time timestamp DEFAULT CURRENT_TIMESTAMP NULL, -- 创建时间
	update_time timestamp DEFAULT CURRENT_TIMESTAMP NULL, -- 最后更新时间
	update_user_id int8 NULL, -- 最后更新人ID，关联 t_sys_user.id
	create_user_id int8 NULL, -- 创建人ID，关联 t_sys_user.id
	"password" varchar(100) DEFAULT ''::character varying NOT NULL, -- 登录密码，BCrypt加密存储
	CONSTRAINT t_sys_user_pkey PRIMARY KEY (id)
);
COMMENT ON TABLE public.t_sys_user IS '系统用户表，存储系统所有用户的基本信息';
